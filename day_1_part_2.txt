#include<bits/stdc++.h>
using namespace std;
int rec(int level,int n)
{
    if(level==n)
        return 1;
    if(level>n)
        return 0;
    int ans=0;
    for(int step=1;step<=3;step++)
    {
        if(level+step<=n)
        {
            int ways=rec(level+step,n);
            ans+=ways;
        }
    }
    return ans;
}
